     │         Execute plugin logic.                                                                                   │

     │                                                                                                                 │
     │         Returns:                                                                                                │
     │             {                                                                                                   │
     │                 "status": "success" | "error",                                                                  │
     │                 "trace_id": str,  # ULID                                                                        │
     │                 "result": Any,    # Plugin-specific output                                                      │
     │                 "error_message": str | None                                                                     │
     │             }                                                                                                   │
     │         """                                                                                                     │
